Code:

var benchmark = $.benchmark(), i = 0;

for(; i < 1000; i++ ) {
  $('b').html();
}

var resultInMs = benchmark.end().result();

$('#1').html("Runtime: " + resultInMs + " ms <br><br>Console output:<br>" + $.benchmark.output());

  // n = 'hej' every time and i will do the current number in the loop

$.benchmark.startTest("name of the test");

for(var i = 0; i < 1000; i++) {
  $.benchmark.start("show hide");
  $('div').hide();
  $('div').show();
  $.benchmark.end("show hide");
}

for(var i = 0; i < 1000; i++) {
  $.benchmark.start("html");
  $('div').html();
  $.benchmark.end("html");
}

var runTime = $.benchmark.endTest("name of the test").result();

$('#4').html("Runtime: " + runTime + " ms <br><br>Console output:<br>" + $.benchmark.output());

var testresult = $.benchmark.start("hide-test").disable().test(function() {

  $("a").hide();

}).result();

var runTime = $.benchmark.enable().end("hide-test", true).result();
